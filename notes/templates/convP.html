Конвертер imgToPDF Python

Данный скрипт считывает все файлы из заданной папки, обрабатывает изображения (меняет яркость и контрастность), затем сохранят все обработанные изображения в другую папку. Далле он все конвертирует в PDF и склеивает. На выходе получается PDF файл из улучшенных изображений. 

from PIL import Image, ImageEnhance
        import os
        import img2pdf

        photos2 = []
        photos = os.listdir("temp_photos")

        for i in range(len(photos)):
            im = Image.open(f"temp_photos/{photos[i]}")
            enhancer = ImageEnhance.Contrast(im)
            factor_contrast = 1.75   # Коэффициент контрастности
            im_output = enhancer.enhance(factor_contrast)
            im_output.save(f"temp_photos2/image_new{i}.jpg")

        photos = os.listdir("temp_photos2")

        for i in range(len(photos)):
            im = Image.open(f"temp_photos2/{photos[i]}")
            enhancer = ImageEnhance.Brightness(im)
            factor_brightness = 1.25   # Коэффициент контрастности
            im_output = enhancer.enhance(factor_brightness)
            im_output.save(f"result_photos/image_new{i}.jpg")
            photos2.append(f"result_photos/image_new{i}.jpg")

        a4_page_size = [img2pdf.in_to_pt(8.3), img2pdf.in_to_pt(11.0)]
        layout_function = img2pdf.get_layout_fun(a4_page_size)

        pdf = img2pdf.convert(*photos2, layout_fun=layout_function)
        with open('A4_doc.pdf', 'wb') as f:
             f.write(pdf)
